#include <iostream>
#include <cmath>
using namespace std;
class Vector {
private:
    double x;
    double y;
public:
    Vector(){};
    Vector(double x, double y){
        this->x=x;
        this->y=y;
    }
    float mod() {                 //модуль
        return sqrt(x * x + y * y);
    };
    void sum(Vector b) {
        x += b.x;
        y += b.y;
    };
   
    };
    void show() {
        cout << x << " " << y << endl;
    }
};
int main(){
    Vector c(90,67);
    Vector d(47,58);
    c.sum(d);
    c.show();
    return 0;
}
